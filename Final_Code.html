Model¶

In [8]:
X = tf.placeholder(tf.float32, [None, len(train_data[1])])
Y = tf.placeholder(tf.float32, [None, 2])

W1 = tf.Variable(tf.random_normal([len(train_data[1]), 1000], stddev=0.01))
L1 = tf.sigmoid(tf.matmul(X, W1))

W2 = tf.Variable(tf.random_normal([1000, 1000], stddev=0.01))
L2 = tf.sigmoid(tf.matmul(L1, W2))

W3 = tf.Variable(tf.random_normal([1000, 2], stddev=0.01))
model = tf.nn.softmax(tf.matmul(L2, W3))

cost = tf.reduce_mean(-tf.reduce_sum(Y * tf.log(model), reduction_indices=[1]))
optimizer = tf.train.AdamOptimizer(0.001).minimize(cost)

Training¶

In [16]:
epochs=200
init = tf.global_variables_initializer()
sess = tf.Session()
sess.run(init)

for epoch in range(epochs):
    total_cost = 0

    _, cost_val = sess.run([optimizer, cost], feed_dict={X: train_data , Y: train_label})
    total_cost += cost_val
    is_correct = tf.equal(tf.argmax(model, 1), tf.argmax(Y, 1))
    accuracy = tf.reduce_mean(tf.cast(is_correct, tf.float32))

    if epoch%10==0:
        print('Epoch:', '%03d' % (epoch + 1), 'error:', '%09f' % (total_cost),'training 정확도:', 100*sess.run(accuracy,
                            feed_dict={X:train_data,
                                       Y: train_label}),'%','test 정확도:', 100*sess.run(accuracy,
                            feed_dict={X:test_data,
                                       Y: test_label}),'%')

print('최적화 완료!')
